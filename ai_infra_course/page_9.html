示例1：数据处理函数生成

需求描述
"创建一个函数，读取CSV文件并计算每列的平均值"
AI生成代码
import pandas as pd

def calculate_column_averages(csv_file_path):
    """读取CSV文件并计算每列的平均值"""
    try:
        df = pd.read_csv(csv_file_path)
        numeric_columns = df.select_dtypes(
            include=['number'])
        averages = numeric_columns.mean().to_dict()
        return averages
    except Exception as e:
        print(f"Error: {e}")
        return None

示例2：算法优化

优化前
# 嵌套循环实现矩阵乘法 O(n³)
def matrix_multiply(A, B):
    result = []
    for i in range(len(A)):
        row = []
        for j in range(len(B[0])):
            sum_val = 0
            for k in range(len(B)):
                sum_val += A[i][k] * B[k][j]
            row.append(sum_val)
        result.append(row)
    return result
AI优化后
import numpy as np

def matrix_multiply_optimized(A, B):
    # 使用NumPy向量化操作
    return np.dot(A, B)

示例3：Bug修复

错误代码
def get_last_element(arr):
    # 数组越界风险
    return arr[len(arr)]  # ❌ 错误!
AI诊断
索引越界：应使用 len(arr)-1
修复后代码
def get_last_element(arr):
    # 添加边界检查
    if not arr:
        raise ValueError("数组为空")
    return arr[len(arr) - 1]  # ✅ 正确!
